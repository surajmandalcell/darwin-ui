import { Tabs, TabItem } from '@astrojs/starlight/components';

The Window component is perfect for code snippets, terminal simulations, or showcasing app interfaces.

## Installation

```tsx
import { Window, WindowHeader, WindowContent } from '@smc/darwin-ui';
```

## Usage

```tsx
export function Example() {
  return (
    <Window className="max-w-[500px]">
      <WindowHeader title="terminal" />
      <WindowContent>
        <div className="font-mono text-sm text-white/80">
          <p>$ npm install</p>
          <p className="text-emerald-400">Success!</p>
        </div>
      </WindowContent>
    </Window>
  );
}
```
