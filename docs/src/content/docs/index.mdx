## Quick Start

```bash
# Install the package
npm install @smc/darwin-ui

# Or add individual components
npx shadcn add https://darwin-ui.mandalsuraj.com/registry/button.json
```

```tsx
import { Button, Card, CardHeader, CardTitle, CardContent } from '@smc/darwin-ui';
import '@smc/darwin-ui/styles';

function App() {
  return (
    <Card>
      <CardHeader>
        <CardTitle>Welcome to Darwin UI</CardTitle>
      </CardHeader>
      <CardContent>
        <Button variant="primary">Get Started</Button>
      </CardContent>
    </Card>
  );
}
```
