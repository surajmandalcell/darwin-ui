import { Tabs, TabItem } from '@astrojs/starlight/components';

This hook provides a simple way to bind a callback to the `Escape` key.

## Installation

<Tabs>
  <TabItem label="Manual">
    ```bash
    npm install @smc/darwin-ui
    ```
  </TabItem>
</Tabs>

## Usage

```tsx
import { useEscapeKey } from '@smc/darwin-ui';

export function Modal() {
  const close = () => console.log('closed');
  useEscapeKey(close);

  return <div>Modal Content</div>;
}
```

## API Reference

```tsx
function useEscapeKey(handler: (event: KeyboardEvent) => void): void
```
