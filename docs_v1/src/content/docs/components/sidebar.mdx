import { Tabs, TabItem } from '@astrojs/starlight/components';

The Sidebar component provides a solid foundation for complex application navigation patterns.

## Installation

```tsx
import { Sidebar, SidebarHeader, SidebarContent, SidebarFooter, SidebarItem } from '@smc/darwin-ui';
```

## Usage

```tsx
export function Example() {
  return (
    <div className="flex bg-[#0a0a0a] border border-white/10 rounded-xl h-[500px] overflow-hidden">
      <Sidebar>
        <SidebarHeader>
           <h3 className="font-bold">App Name</h3>
        </SidebarHeader>
        <SidebarContent>
           <SidebarItem active icon={<Home />}>Dashboard</SidebarItem>
           <SidebarItem icon={<Users />}>Team</SidebarItem>
           <SidebarItem icon={<Settings />}>Settings</SidebarItem>
        </SidebarContent>
        <SidebarFooter>
           <div className="text-sm text-white/50">User Profile</div>
        </SidebarFooter>
      </Sidebar>
      <div className="flex-1 p-8">
        <h1>Content Area</h1>
      </div>
    </div>
  );
}
```
